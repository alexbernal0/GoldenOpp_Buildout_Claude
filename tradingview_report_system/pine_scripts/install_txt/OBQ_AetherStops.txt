//@version=6
indicator("OBQ Aether Stops", shorttitle="OBQ-AS", overlay=true,
          max_lines_count=500, max_labels_count=100, max_boxes_count=10)

// ════════════════════════════════════════════════════════════════════════════════
// OBQ AETHER STOPS — Pine Script v6 port of the NinjaTrader Aether Stops tool
// by Affordable Indicators, adapted for TradingView.
//
// HOW TO USE:
//   1. Add indicator to chart
//   2. Open Settings (gear icon) → set "Anchor Date" to the bar you want to
//      anchor the stops to (the bar whose high/low becomes the starting point)
//   3. Set Direction: Long (stop below) or Short (stop above)
//   4. Enable the stop types you want — each has its own parameters
//   5. Toggle "Active" off to clear all stops
//
// STOP TYPES:
//   S1 Fixed Ticks  — fixed number of ticks from the anchor high/low
//   S2 Fixed %      — fixed percentage of price from anchor
//   S3 ATR          — ATR(period) × multiplier from anchor, trailing
//   S4 WATR         — Weighted ATR (WMA of High-Low) × multiplier, trailing
//
// PROFIT TARGETS (Long only — ATR-based, static from anchor):
//   T1 / T2 / T3    — ATR × multiplier above anchor high
//
// TRAIL LOGIC: Each stop ratchets with price (Long: anchors to the highest high
// seen since anchor bar; Short: anchors to lowest low). Line terminates when
// price hits the stop level (if "Terminate When Hit" is enabled).
// ════════════════════════════════════════════════════════════════════════════════


// ── SECTION 1: MASTER CONTROLS ───────────────────────────────────────────────

grp_ctrl = "⚙  Controls"
active    = input.bool(true,  "Active",         group=grp_ctrl,
                       tooltip="Turn off to hide all stops and clear the chart")
direction = input.string("Long",  "Direction",   group=grp_ctrl,
                         options=["Long","Short"],
                         tooltip="Long = stop below price (trailing up)\nShort = stop above price (trailing down)")
anc_time  = input.time(timestamp("2026-03-20 00:00 +0000"), "Anchor Date",
                       group=grp_ctrl,
                       tooltip="Set this to the date of the bar you want to anchor stops to.\nUse the calendar picker — choose the exact candle date.")
ibm       = input.string("Low/High", "Anchor Mode", group=grp_ctrl,
                         options=["Low/High","Close"],
                         tooltip="Low/High: anchor to the bar's Low (Long) or High (Short)\nClose: anchor to the bar's Close for both directions")
terminate = input.bool(true, "Terminate When Hit", group=grp_ctrl,
                       tooltip="Stop line ends at the bar where price crosses the stop level")
show_tbl  = input.bool(true, "Show Info Panel",   group=grp_ctrl)


// ── SECTION 2: STOP 1 — FIXED TICKS ─────────────────────────────────────────

grp_s1    = "\tS1  Fixed Ticks"
s1_on     = input.bool(true,  "Enabled",      group=grp_s1)
s1_ticks  = input.float(15.0, "Ticks",        group=grp_s1, minval=0.0,
                        tooltip="Number of ticks (instrument tick size × this value) below/above anchor")
s1_col    = input.color(color.new(#0D47A1, 0), "Color", group=grp_s1)
s1_width  = input.int(1, "Width",             group=grp_s1, minval=1, maxval=5)


// ── SECTION 3: STOP 2 — FIXED PERCENT ────────────────────────────────────────

grp_s2    = "\tS2  Fixed %"
s2_on     = input.bool(true,  "Enabled",      group=grp_s2)
s2_pct    = input.float(0.5,  "Percent",      group=grp_s2, minval=0.0, step=0.1,
                        tooltip="Percent of current close to use as stop distance (e.g. 0.5 = 0.5%)")
s2_col    = input.color(color.new(#1565C0, 0), "Color", group=grp_s2)
s2_width  = input.int(1, "Width",             group=grp_s2, minval=1, maxval=5)


// ── SECTION 4: STOP 3 — ATR TRAILING ─────────────────────────────────────────

grp_s3    = "\tS3  ATR Trailing"
s3_on     = input.bool(true,  "Enabled",      group=grp_s3)
s3_per    = input.int(14,     "ATR Period",   group=grp_s3, minval=1)
s3_mult   = input.float(2.5,  "Multiplier",   group=grp_s3, minval=0.1, step=0.5,
                        tooltip="Stop = Highest High since anchor − ATR(period) × multiplier  (Long)")
s3_col    = input.color(color.new(#212121, 0), "Color", group=grp_s3)
s3_width  = input.int(1, "Width",             group=grp_s3, minval=1, maxval=5)


// ── SECTION 5: STOP 4 — WATR TRAILING ────────────────────────────────────────

grp_s4    = "\tS4  WATR Trailing"
s4_on     = input.bool(true,  "Enabled",      group=grp_s4)
s4_per    = input.int(14,     "WATR Period",  group=grp_s4, minval=1,
                        tooltip="WATR = WMA of True Range (High−Low). Smoother than ATR, ignores gaps.")
s4_mult   = input.float(2.0,  "Multiplier",   group=grp_s4, minval=0.1, step=0.5)
s4_col    = input.color(color.new(#263238, 0), "Color", group=grp_s4)
s4_width  = input.int(1, "Width",             group=grp_s4, minval=1, maxval=5)


// ── SECTION 6: PROFIT TARGETS (Long only) ────────────────────────────────────

grp_tgt   = "\tTargets  (Long only)"
t_per     = input.int(14,    "ATR Period",    group=grp_tgt, minval=1,
                       tooltip="ATR period used for all three targets")
t1_mult   = input.float(2.0, "T1 Multiplier", group=grp_tgt, minval=0.0, step=0.5,
                       tooltip="Set to 0 to disable")
t2_mult   = input.float(3.0, "T2 Multiplier", group=grp_tgt, minval=0.0, step=0.5,
                       tooltip="Set to 0 to disable")
t3_mult   = input.float(4.0, "T3 Multiplier", group=grp_tgt, minval=0.0, step=0.5,
                       tooltip="Set to 0 to disable")
t_col     = input.color(color.new(#66BB6A, 50), "Color",  group=grp_tgt)
t_width   = input.int(1, "Width",               group=grp_tgt, minval=1, maxval=5)


// ── COMPUTED VALUES (every bar) ───────────────────────────────────────────────

// ATR and WATR — computed on every bar (Pine v6 requirement)
atr_s3 = ta.atr(s3_per)
atr_s4 = ta.wma(high - low, s4_per)      // WATR: WMA of High-Low range
atr_t  = ta.atr(t_per)

is_long = direction == "Long"

// ── ANCHOR LOOKUP (resolved on last bar via full history scan) ────────────────
// We scan backwards from the current bar to find the bar whose daily timestamp
// best matches anc_time. Done entirely on barstate.islast — same approach as
// the Forward Cloud. Robust against input changes and daily/intraday charts.

// ── ANCHOR DETECTION ─────────────────────────────────────────────────────────
// Track the last bar that falls within a 2-day window of anc_time.
// We update anc_bar on every bar in the window — the last update will be
// the bar closest to (but not past) anc_time. No na() guard needed.
var int   anc_bar   = na
var float anc_price = na
var float anc_atr3  = na
var float anc_atr4  = na
var float anc_atrt  = na

// Window: bars whose timestamp is within 2 days before or after anc_time
bool in_anc_window = time >= anc_time - 172800000 and time <= anc_time + 172800000

if in_anc_window
    anc_bar   := bar_index
    anc_price := ibm == "Close" ? close : (is_long ? low : high)
    anc_atr3  := atr_s3
    anc_atr4  := atr_s4
    anc_atrt  := atr_t


// ── DRAW ALL STOPS ON LAST BAR ────────────────────────────────────────────────
// We redraw from scratch on every barstate.islast (same as Forward Cloud pattern).
// This avoids line accumulation and max_lines_count issues.

// Persistent line arrays — cleared and redrawn each last bar
var line[] s1_lines = array.new_line()
var line[] s2_lines = array.new_line()
var line[] s3_lines = array.new_line()
var line[] s4_lines = array.new_line()
var line[] t1_lines = array.new_line()
var line[] t2_lines = array.new_line()
var line[] t3_lines = array.new_line()
var label[] anc_labels = array.new_label()

// Helper: delete all lines in an array and clear it
clear_lines(line[] arr) =>
    int n = array.size(arr)
    if n > 0
        for i = 0 to n - 1
            line.delete(array.get(arr, i))
    array.clear(arr)

// Helper: delete all labels in an array and clear it
clear_labels(label[] arr) =>
    int n = array.size(arr)
    if n > 0
        for i = 0 to n - 1
            label.delete(array.get(arr, i))
    array.clear(arr)

// ── Info panel state (updated on last bar for display) ────────────────────────
var float info_s1 = na
var float info_s2 = na
var float info_s3 = na
var float info_s4 = na
var float info_t1 = na
var float info_t2 = na
var float info_t3 = na
var bool  info_any_hit = false
var int   info_bars_active = 0

if barstate.islast

    // ── Clear all previous drawings ─────────────────────────────────────────
    clear_lines(s1_lines)
    clear_lines(s2_lines)
    clear_lines(s3_lines)
    clear_lines(s4_lines)
    clear_lines(t1_lines)
    clear_lines(t2_lines)
    clear_lines(t3_lines)
    clear_labels(anc_labels)

    info_s1 := na
    info_s2 := na
    info_s3 := na
    info_s4 := na
    info_t1 := na
    info_t2 := na
    info_t3 := na
    info_any_hit := false

    if active and not na(anc_bar) and anc_bar <= bar_index

        int total_bars = bar_index - anc_bar
        info_bars_active := total_bars

        // Draw anchor bar marker
        float anc_hi = high[total_bars]
        float anc_lo = low[total_bars]
        label.new(anc_bar, is_long ? anc_lo : anc_hi,
                  is_long ? "^ Anchor" : "v Anchor",
                  style     = is_long ? label.style_label_up : label.style_label_down,
                  color     = color.new(#37474F, 20),
                  textcolor = color.white,
                  size      = size.small,
                  xloc      = xloc.bar_index)
        array.push(anc_labels, label.new(anc_bar, is_long ? anc_lo : anc_hi, ""))
        // (second push is dummy to keep array non-empty — first label is the real one)

        // ── Walk each bar from anchor to now, computing stop levels ────────
        // Long:  anchor_price = low[total_bars], trail = highest high
        // Short: anchor_price = high[total_bars], trail = lowest low

        float running_peak   = anc_price
        float running_trough = anc_price

        // Per-stop termination flags
        bool s1_hit = false
        bool s2_hit = false
        bool s3_hit = false
        bool s4_hit = false
        bool t1_hit = false
        bool t2_hit = false
        bool t3_hit = false

        float prev_s1 = na
        float prev_s2 = na
        float prev_s3 = na
        float prev_s4 = na
        float prev_t1 = na
        float prev_t2 = na
        float prev_t3 = na
        int   prev_bi = na

        for offset = total_bars to 0
            int   bi    = anc_bar + (total_bars - offset)  // absolute bar_index
            float h     = high[offset]
            float l     = low[offset]
            float c     = close[offset]
            float atr3v = atr_s3[offset]
            float atr4v = atr_s4[offset]
            float atrtv = atr_t[offset]

            // Update running peak/trough
            if offset == total_bars
                running_peak   := anc_price
                running_trough := anc_price
            else
                running_peak   := math.max(running_peak,   h)
                running_trough := math.min(running_trough, l)

            // ── Compute stop levels at this bar ─────────────────────────────
            float s1v = na
            float s2v = na
            float s3v = na
            float s4v = na
            float t1v = na
            float t2v = na
            float t3v = na

            if is_long
                // S1 Fixed Ticks
                if s1_on and not s1_hit
                    s1v := running_peak - s1_ticks * syminfo.mintick

                // S2 Fixed %
                if s2_on and not s2_hit
                    s2v := running_peak - running_peak * s2_pct / 100.0

                // S3 ATR
                if s3_on and not s3_hit
                    s3v := running_peak - atr3v * s3_mult

                // S4 WATR
                if s4_on and not s4_hit
                    s4v := running_peak - atr4v * s4_mult

                // Targets — static from anchor, only computed once (use anchor ATR)
                if t1_mult > 0 and not t1_hit
                    t1v := anc_price + anc_atrt * t1_mult
                if t2_mult > 0 and not t2_hit
                    t2v := anc_price + anc_atrt * t2_mult
                if t3_mult > 0 and not t3_hit
                    t3v := anc_price + anc_atrt * t3_mult

            else
                // SHORT
                if s1_on and not s1_hit
                    s1v := running_trough + s1_ticks * syminfo.mintick
                if s2_on and not s2_hit
                    s2v := running_trough + running_trough * s2_pct / 100.0
                if s3_on and not s3_hit
                    s3v := running_trough + atr3v * s3_mult
                if s4_on and not s4_hit
                    s4v := running_trough + atr4v * s4_mult
                // No targets for short (consistent with original)

            // ── Check if stop was hit on this bar ───────────────────────────
            if terminate and offset < total_bars  // don't terminate on anchor bar
                if is_long
                    if s1_on and not s1_hit and not na(s1v) and l <= s1v
                        s1_hit := true
                        info_any_hit := true
                    if s2_on and not s2_hit and not na(s2v) and l <= s2v
                        s2_hit := true
                        info_any_hit := true
                    if s3_on and not s3_hit and not na(s3v) and l <= s3v
                        s3_hit := true
                        info_any_hit := true
                    if s4_on and not s4_hit and not na(s4v) and l <= s4v
                        s4_hit := true
                        info_any_hit := true
                    if t1_mult > 0 and not t1_hit and not na(t1v) and h >= t1v
                        t1_hit := true
                    if t2_mult > 0 and not t2_hit and not na(t2v) and h >= t2v
                        t2_hit := true
                    if t3_mult > 0 and not t3_hit and not na(t3v) and h >= t3v
                        t3_hit := true
                else
                    if s1_on and not s1_hit and not na(s1v) and h >= s1v
                        s1_hit := true
                        info_any_hit := true
                    if s2_on and not s2_hit and not na(s2v) and h >= s2v
                        s2_hit := true
                        info_any_hit := true
                    if s3_on and not s3_hit and not na(s3v) and h >= s3v
                        s3_hit := true
                        info_any_hit := true
                    if s4_on and not s4_hit and not na(s4v) and h >= s4v
                        s4_hit := true
                        info_any_hit := true

            // ── Draw line segments bar by bar ───────────────────────────────
            if not na(prev_bi)
                if not na(prev_s1) and not na(s1v) and s1_on
                    array.push(s1_lines, line.new(prev_bi, prev_s1, bi, s1v,
                                                  color=s1_col, width=s1_width, xloc=xloc.bar_index))
                if not na(prev_s2) and not na(s2v) and s2_on
                    array.push(s2_lines, line.new(prev_bi, prev_s2, bi, s2v,
                                                  color=s2_col, width=s2_width, xloc=xloc.bar_index))
                if not na(prev_s3) and not na(s3v) and s3_on
                    array.push(s3_lines, line.new(prev_bi, prev_s3, bi, s3v,
                                                  color=s3_col, width=s3_width, xloc=xloc.bar_index))
                if not na(prev_s4) and not na(s4v) and s4_on
                    array.push(s4_lines, line.new(prev_bi, prev_s4, bi, s4v,
                                                  color=s4_col, width=s4_width, xloc=xloc.bar_index))
                if is_long
                    if not na(prev_t1) and not na(t1v) and t1_mult > 0
                        array.push(t1_lines, line.new(prev_bi, prev_t1, bi, t1v,
                                                      color=t_col, width=t_width, style=line.style_dashed, xloc=xloc.bar_index))
                    if not na(prev_t2) and not na(t2v) and t2_mult > 0
                        array.push(t2_lines, line.new(prev_bi, prev_t2, bi, t2v,
                                                      color=t_col, width=t_width, style=line.style_dashed, xloc=xloc.bar_index))
                    if not na(prev_t3) and not na(t3v) and t3_mult > 0
                        array.push(t3_lines, line.new(prev_bi, prev_t3, bi, t3v,
                                                      color=t_col, width=t_width, style=line.style_dashed, xloc=xloc.bar_index))

            // After a stop is hit, set prev to na so no more segments draw
            prev_s1 := s1_hit ? na : s1v
            prev_s2 := s2_hit ? na : s2v
            prev_s3 := s3_hit ? na : s3v
            prev_s4 := s4_hit ? na : s4v
            prev_t1 := t1_hit ? na : t1v
            prev_t2 := t2_hit ? na : t2v
            prev_t3 := t3_hit ? na : t3v
            prev_bi := bi

        // ── Capture current stop levels for info panel ──────────────────────
        info_s1 := prev_s1
        info_s2 := prev_s2
        info_s3 := prev_s3
        info_s4 := prev_s4
        info_t1 := prev_t1
        info_t2 := prev_t2
        info_t3 := prev_t3

        // ── End-of-line labels ──────────────────────────────────────────────
        // Draw a small label at the right end of each active stop line
        int last_bi = bar_index
        if s1_on and not na(info_s1)
            array.push(anc_labels, label.new(last_bi, info_s1,
                       "S1 " + str.tostring(math.round(info_s1, 2)),
                       style=label.style_label_left, xloc=xloc.bar_index,
                       color=color.new(s1_col, 70), textcolor=color.white, size=size.tiny))
        if s2_on and not na(info_s2)
            array.push(anc_labels, label.new(last_bi, info_s2,
                       "S2 " + str.tostring(math.round(info_s2, 2)),
                       style=label.style_label_left, xloc=xloc.bar_index,
                       color=color.new(s2_col, 70), textcolor=color.white, size=size.tiny))
        if s3_on and not na(info_s3)
            array.push(anc_labels, label.new(last_bi, info_s3,
                       "S3 " + str.tostring(math.round(info_s3, 2)),
                       style=label.style_label_left, xloc=xloc.bar_index,
                       color=color.new(s3_col, 70), textcolor=color.white, size=size.tiny))
        if s4_on and not na(info_s4)
            array.push(anc_labels, label.new(last_bi, info_s4,
                       "S4 " + str.tostring(math.round(info_s4, 2)),
                       style=label.style_label_left, xloc=xloc.bar_index,
                       color=color.new(s4_col, 70), textcolor=color.white, size=size.tiny))
        if is_long
            if t1_mult > 0 and not na(info_t1)
                array.push(anc_labels, label.new(last_bi, info_t1,
                           "T1 " + str.tostring(math.round(info_t1, 2)),
                           style=label.style_label_left, xloc=xloc.bar_index,
                           color=color.new(t_col, 40), textcolor=color.white, size=size.tiny))
            if t2_mult > 0 and not na(info_t2)
                array.push(anc_labels, label.new(last_bi, info_t2,
                           "T2 " + str.tostring(math.round(info_t2, 2)),
                           style=label.style_label_left, xloc=xloc.bar_index,
                           color=color.new(t_col, 40), textcolor=color.white, size=size.tiny))
            if t3_mult > 0 and not na(info_t3)
                array.push(anc_labels, label.new(last_bi, info_t3,
                           "T3 " + str.tostring(math.round(info_t3, 2)),
                           style=label.style_label_left, xloc=xloc.bar_index,
                           color=color.new(t_col, 40), textcolor=color.white, size=size.tiny))


// ── INFO PANEL (table in top-right corner) ────────────────────────────────────

var table info_tbl = na

if barstate.islast and show_tbl
    if na(info_tbl)
        info_tbl := table.new(position.top_right, 2, 12,
                              bgcolor       = color.new(#1A1A2E, 10),
                              border_color  = color.new(#37474F, 50),
                              border_width  = 1,
                              frame_color   = color.new(#37474F, 30),
                              frame_width   = 1)

    color hdr_bg   = color.new(#1565C0, 10)
    color hdr_txt  = color.white
    color val_txt  = color.new(#CFD8DC, 0)
    color hit_col  = color.new(#EF5350, 0)
    color ok_col   = color.new(#26A69A, 0)
    color dim_col  = color.new(#546E7A, 0)
    string fnt     = size.small

    // Header
    table.cell(info_tbl, 0, 0, "OBQ Aether Stops",
               text_color=hdr_txt, bgcolor=hdr_bg,
               text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 0,
               (active ? (is_long ? "^ LONG" : "v SHORT") : "OFF"),
               text_color = active ? (is_long ? ok_col : hit_col) : dim_col,
               bgcolor=hdr_bg, text_size=fnt, text_halign=text.align_right)

    // Anchor info
    string anc_str = na(anc_bar) ? "No anchor set" :
                     "Bar " + str.tostring(anc_bar) +
                     "  (" + str.tostring(info_bars_active) + " bars ago)"
    table.cell(info_tbl, 0, 1, "Anchor", text_color=dim_col, bgcolor=color.new(#0D1B2A, 30), text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 1, anc_str,  text_color=val_txt, bgcolor=color.new(#0D1B2A, 30), text_size=fnt, text_halign=text.align_right)

    // Anchor price
    string anc_price_str = na(anc_price) ? "—" : str.tostring(math.round(anc_price, 2))
    table.cell(info_tbl, 0, 2, "Anchor Price", text_color=dim_col, bgcolor=color.new(#0D1B2A, 50), text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 2, anc_price_str,  text_color=val_txt, bgcolor=color.new(#0D1B2A, 50), text_size=fnt, text_halign=text.align_right)

    // S1
    string s1_str = not s1_on ? "disabled" : na(info_s1) ? "[HIT]" : str.tostring(math.round(info_s1, 2))
    color  s1_c   = not s1_on ? dim_col : na(info_s1) ? hit_col : ok_col
    table.cell(info_tbl, 0, 3, "S1 Fixed Ticks (" + str.tostring(s1_ticks) + ")", text_color=dim_col, text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 3, s1_str, text_color=s1_c, text_size=fnt, text_halign=text.align_right)

    // S2
    string s2_str = not s2_on ? "disabled" : na(info_s2) ? "[HIT]" : str.tostring(math.round(info_s2, 2))
    color  s2_c   = not s2_on ? dim_col : na(info_s2) ? hit_col : ok_col
    table.cell(info_tbl, 0, 4, "S2 Fixed % (" + str.tostring(s2_pct) + "%)", text_color=dim_col, text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 4, s2_str, text_color=s2_c, text_size=fnt, text_halign=text.align_right)

    // S3
    string s3_str = not s3_on ? "disabled" : na(info_s3) ? "[HIT]" : str.tostring(math.round(info_s3, 2))
    color  s3_c   = not s3_on ? dim_col : na(info_s3) ? hit_col : ok_col

    string s4_str = not s4_on ? "disabled" : na(info_s4) ? "[HIT]" : str.tostring(math.round(info_s4, 2))
    color  s4_c   = not s4_on ? dim_col : na(info_s4) ? hit_col : ok_col

    string t1_str = t1_mult == 0 ? "disabled" : na(info_t1) ? "[HIT]" : str.tostring(math.round(info_t1, 2))

    color  t3_c   = t3_mult == 0 ? dim_col : na(info_t3) ? hit_col : ok_col
    table.cell(info_tbl, 0, 8,  "T1 ATR x" + str.tostring(t1_mult), text_color=dim_col, text_size=fnt, text_halign=text.align_left)

    table.cell(info_tbl, 0, 10, "T3 ATR x" + str.tostring(t3_mult), text_color=dim_col, text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 10, t3_str, text_color=t3_c, text_size=fnt, text_halign=text.align_right)

    // Status footer
    string status  = "[ * ] Active - trailing"
    color  status_c = ok_col
    if not active
        status   := "[ OFF ] Inactive - enable in Settings"
        status_c := dim_col
    else if na(anc_bar)
        status   := "[ ! ] Set Anchor Date in Settings"
        status_c := color.new(#FF9800, 0)
    else if info_any_hit

        status_c := hit_col
    table.cell(info_tbl, 0, 11, status,   text_color=status_c, text_size=fnt, text_halign=text.align_left)
    table.cell(info_tbl, 1, 11, "",       text_color=status_c, text_size=fnt, text_halign=text.align_right)
