//@version=6
// OBQ Support & Resistance v2 — Provably Non-Repainting
// All-black lines, thickness = touch count (1=hairline → 5+=solid bold)
// Architecture: pivot-cluster, confirmed on closed bars, drawn at confirmation time
indicator("OBQ S/R Levels v2", overlay=true, max_lines_count=500, max_labels_count=200)

// ─── INPUTS ──────────────────────────────────────────────────────────────────
var g1 = "Detection"
i_leftStr    = input.int(3,     "Pivot Left Bars",     minval=1, maxval=10, group=g1)
i_rightStr   = input.int(3,     "Pivot Right Bars",    minval=1, maxval=10, group=g1,
                 tooltip="Pivot confirmed RIGHT bars after pivot bar. Non-repainting cost.")
i_combinePct = input.float(0.75,"Combine Zone %",      minval=0.1, maxval=3.0, group=g1, step=0.05,
                 tooltip="Pivots within this % of anchor price are merged. Anchor never moves.")
i_lookback   = input.int(500,   "Lookback Bars",       minval=50, maxval=2000, group=g1,
                 tooltip="Levels outside this window are hidden.")

var g2 = "Display"
i_minTouches = input.int(2,     "Min Touches to Show", minval=1, maxval=10, group=g2,
                 tooltip="1-touch = noise. 2+ = confirmed level.")
i_hotAt      = input.int(3,     "Solid at N touches",  minval=2, maxval=20, group=g2,
                 tooltip="3+ = thick solid line. 2 = thin dashed.")
i_maxLines   = input.int(40,    "Max Lines on Chart",  minval=5, maxval=200, group=g2)
i_showLabels = input.bool(true,  "Show Touch Labels",  group=g2)
i_showLag    = input.bool(false, "Show Lag Markers",   group=g2)

// ─── STATE ───────────────────────────────────────────────────────────────────
var float[] lv_price   = array.new<float>()
var int[]   lv_touches = array.new<int>()
var int[]   lv_bar     = array.new<int>()
var line[]  lv_line    = array.new<line>()
var label[] lv_label   = array.new<label>()

// ─── VISUAL ENCODING — all black, thickness encodes touch count ───────────────
// 1 touch  = width 1, 70% transparent (ghost line)
// 2 touches = width 1, 40% transparent
// 3 touches = width 2, 20% transparent
// 4 touches = width 3, 10% transparent
// 5+ touches = width 4, fully opaque (solid bold black)
line_col(int t) =>
    t >= 5 ? color.new(color.black,  0) :
     t == 4 ? color.new(color.black, 10) :
     t == 3 ? color.new(color.black, 20) :
     t == 2 ? color.new(color.black, 40) :
              color.new(color.black, 70)

line_wid(int t) =>
    t >= 5 ? 4 : t == 4 ? 3 : t == 3 ? 2 : 1

line_sty(int t) =>
    t >= i_hotAt ? line.style_solid : line.style_dashed

// ─── HELPERS ─────────────────────────────────────────────────────────────────
find_level(float price) =>
    result = -1
    tol = price * (i_combinePct / 100.0)
    n = array.size(lv_price)
    if n > 0
        for i = 0 to n - 1
            if math.abs(array.get(lv_price, i) - price) <= tol
                result := i
                break
    result

update_visuals(int idx) =>
    t  = array.get(lv_touches, idx)
    ln = array.get(lv_line,    idx)
    lb = array.get(lv_label,   idx)
    if not na(ln)
        line.set_color(ln, line_col(t))
        line.set_width(ln, line_wid(t))
        line.set_style(ln, line_sty(t))
    if not na(lb) and i_showLabels
        label.set_text(lb, str.tostring(t) + "x")
        label.set_textcolor(lb, line_col(t))

create_level(float price, int confirm_bar) =>
    pivot_bar = bar_index - i_rightStr
    t = 1
    ln = line.new(
         x1    = pivot_bar,
         y1    = price,
         x2    = bar_index,
         y2    = price,
         xloc  = xloc.bar_index,
         extend= extend.right,
         color = line_col(t),
         width = line_wid(t),
         style = line_sty(t))
    lb = i_showLabels ?
         label.new(
             x         = bar_index,
             y         = price,
             text      = "1x",
             xloc      = xloc.bar_index,
             color     = color.new(color.black, 100),
             textcolor = line_col(t),
             style     = label.style_label_left,
             size      = size.tiny) : na
    array.push(lv_price,   price)
    array.push(lv_touches, 1)
    array.push(lv_bar,     confirm_bar)
    array.push(lv_line,    ln)
    array.push(lv_label,   lb)

process_pivot(float pivot_price) =>
    idx = find_level(pivot_price)
    if idx >= 0
        array.set(lv_touches, idx, array.get(lv_touches, idx) + 1)
        update_visuals(idx)
    else
        create_level(pivot_price, bar_index)

prune_old_levels() =>
    cutoff = bar_index - i_lookback
    i = array.size(lv_price) - 1
    while i >= 0
        if array.get(lv_bar, i) < cutoff
            ln = array.get(lv_line,  i)
            lb = array.get(lv_label, i)
            if not na(ln)
                line.delete(ln)
            if not na(lb)
                label.delete(lb)
            array.remove(lv_price,   i)
            array.remove(lv_touches, i)
            array.remove(lv_bar,     i)
            array.remove(lv_line,    i)
            array.remove(lv_label,   i)
        i -= 1

enforce_max_lines() =>
    n = array.size(lv_price)
    if n > i_maxLines
        for i = 0 to n - 1
            t  = array.get(lv_touches, i)
            ln = array.get(lv_line, i)
            lb = array.get(lv_label, i)
            if t < i_minTouches and not na(ln)
                line.set_color(ln, color.new(color.black, 100))
                if not na(lb)
                    label.set_color(lb, color.new(color.black, 100))

// ─── MAIN LOGIC ──────────────────────────────────────────────────────────────
if barstate.isconfirmed
    float ph = ta.pivothigh(high, i_leftStr, i_rightStr)
    float pl = ta.pivotlow(low,   i_leftStr, i_rightStr)

    if not na(ph)
        process_pivot(ph)
        if i_showLag
            label.new(bar_index - i_rightStr, ph, text="H",
                     color=color.new(color.teal, 70), textcolor=color.white,
                     style=label.style_circle, size=size.tiny)

    if not na(pl)
        process_pivot(pl)
        if i_showLag
            label.new(bar_index - i_rightStr, pl, text="L",
                     color=color.new(color.maroon, 70), textcolor=color.white,
                     style=label.style_circle, size=size.tiny)

    if bar_index % 10 == 0
        prune_old_levels()
        enforce_max_lines()

// ─── LABEL POSITIONS (cosmetic only) ─────────────────────────────────────────
if barstate.islast and i_showLabels
    n = array.size(lv_price)
    if n > 0
        for i = 0 to n - 1
            t  = array.get(lv_touches, i)
            lb = array.get(lv_label,   i)
            if not na(lb) and t >= i_minTouches
                label.set_x(lb, bar_index)

// ─── BACKGROUND PULSE near strong levels ─────────────────────────────────────
var bool near_hot = false
if barstate.islast
    cp = close
    n  = array.size(lv_price)
    near_hot := false
    if n > 0
        for i = 0 to n - 1
            t  = array.get(lv_touches, i)
            lp = array.get(lv_price,   i)
            if t >= i_hotAt and math.abs(lp - cp) / cp < 0.005
                near_hot := true
                break

bgcolor(near_hot ? color.new(color.black, 92) : na, title="Near Strong Level")
